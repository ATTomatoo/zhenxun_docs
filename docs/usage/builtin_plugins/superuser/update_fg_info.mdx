import Messenger from "/src/components/Messenger"

### 说明

更新群组/好友信息

- 插件类型: `SUPERUSER`
- 响应规则: `全局`
- 插件等级: `5`

### 配置项

无

### 指令

- 更新群组信息

- 更新好友信息

### 示例

<Messenger
  msgs={[
    { position: "right", msg: "真寻更新群组信息" },
    {
      position: "left",
      msg: `已经成功更新了群组信息!`,
    },
    { position: "right", msg: "真寻更新好友信息" },
    {
      position: "left",
      msg: `成功更新了 2 个好友的信息`,
    },
  ]}
/>
